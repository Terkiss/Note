	public void diceNum()
	{
		int loopCount = 10000;
		int randomRange = 6;
		int[] randomTable = new int[randomRange];
		Random random =new Random();
		
		for(int i = 0 ; i < loopCount; i++)
		{
			random.setSeed(System.currentTimeMillis());
			int randview = random.nextInt(randomRange);
			randomTable[randview]++;
		}
		
	}
